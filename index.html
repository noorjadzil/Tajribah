import { APP_CONFIG } from './config.js'; export function formatRp(number) { return `${APP_CONFIG.currency} ${Number(number || 0).toLocaleString('id-ID')}`; } export function makeNotaText(cart) { const total = cart.reduce((sum, item) => sum + item.harga * item.qty, 0); const lines = [ `*NOTA ${APP_CONFIG.appName.toUpperCase()}*`, new Date().toLocaleString('id-ID'), '', ...cart.map(item => `${item.nama} x${item.qty} = ${formatRp(item.harga * item.qty)}`), '', `Total: *${formatRp(total)}*` ]; return lines.join('\n'); } export function openWhatsApp(text) { if (!text || text.includes('Total: *Rp 0*')) return alert('Keranjang masih kosong'); const number = APP_CONFIG.waNumber; const url = number ? `https://wa.me/${number}?text=${encodeURIComponent(text)}` : `https://wa.me/?text=${encodeURIComponent(text)}`; window.open(url, '_blank'); }
